    کارگاه فصل ۳ (دکتر دهقان)

    سوال اول

    برنامه ای بنویسید که یک جدول \(6 \times 6\) چاپ کند

    1.مقدار هر خانه برابر \(|i \times j|\) باشد

    2.اگر مقدار زوج بود، مقدار آن چاپ شود

    3.اگر مقدار فرد بود، به جای آن - چاپ شود

    پاسخ 
    #include <iostream>

using namespace std;

int main() {
    // چاپ یک جدول 6x6 با شرایط داده شده
    cout << "--- جدول 6x6 ---\n";

    // حلقه بیرونی برای سطرها (i از 1 تا 6)
    for (int i = 1; i <= 6; ++i) {
        // حلقه داخلی برای ستون ها (j از 1 تا 6)
        for (int j = 1; j <= 6; ++j) {
            // محاسبه مقدار خانه: |i * j|
            // از آنجایی که i و j همیشه مثبت هستند، می توان از abs صرف نظر کرد.
            int value = i * j;

            // بررسی شرط زوج یا فرد بودن
            if (value % 2 == 0) {
                // اگر مقدار زوج بود، آن را چاپ کن
                cout << setw(4) << value;
            } else {
                // اگر مقدار فرد بود، به جای آن '-' چاپ کن
                cout << setw(4) << "-";
            }
        }
        // رفتن به سطر بعدی پس از اتمام یک سطر
        cout << endl;
    }

    return 0;
}

    سوال دوم

    یک شرکت ۵ شعبه دارد. مدیر سیستم می‌خواهد جدول \(5 \times 5\) بسازد که در آن خانه \((i, j)\) نشان‌دهنده فاصله تقریبی بین شعبه \(i\) و شعبه \(j\) باشد.

    • فاصله بین دو شعبه مقدار هر خانه را برابر با \(Distance(i, j) = |i-j| [cite_start]\times 10\) می‌گیریم.

    نکته ۱: فاصله یک شعبه تا خودش تعریفی ندارد و بنابراین قطرهای این جدول خالی می‌مانند.

    اگر دو شعبه برابر بودند، فاصله آن‌ها به اندازه "\t" خالی باشد، در غیر این صورت فاصله بین دو شعبه محاسبه شود.

    پاسخ 
    #include <iostream>
using namespace std;

int main() {
    const int num_branches = 5;
    cout << "--- جدول فاصله شعب (" << num_branches << "x" << num_branches << ") ---\n";

    // حلقه بیرونی برای سطرها (i از 1 تا 5)
    for (int i = 1; i <= num_branches; ++i) {
        // حلقه داخلی برای ستون ها (j از 1 تا 5)
        for (int j = 1; j <= num_branches; ++j) {
            // بررسی نکته ۱: اگر دو شعبه برابر بودند (فاصله یک شعبه تا خودش)
            if (i == j) {
                // فاصله آنها به اندازه "t" خالی باشد
                cout << setw(6) << "t";
            } else {
                // محاسبه فاصله: Distance(i, j) = |i-j| * 10
                int distance = abs(i - j) * 10;
                // چاپ فاصله محاسبه شده
                cout << setw(6) << distance;
            }
        }
        // رفتن به سطر بعدی
        cout << endl;
    }

    return 0;
}

    سوال سوم

    با استفاده از حلقه for برنامه‌ای بنویسید که اعداد بین ۱ تا ۱۰ را به شرطی که آن عدد زوج باشد، چاپ کند.

    پاسخ 
    #include <iostream>

using namespace std;

int main() {
    cout << "--- اعداد زوج بین 1 تا 10 ---\n";
    // حلقه for برای پیمایش اعداد از 1 تا 10
    for (int num = 1; num <= 10; ++num) {
        // شرط: بررسی زوج بودن عدد
        if (num % 2 == 0) {
            // چاپ عدد در صورت زوج بودن
            cout << num << endl;
        }
    }

    return 0;
}

    سوال چهارم

    برنامه‌ای با حلقه for بنویسید که یک مثلث عددی افزایشی چاپ کند.

    این مثلث باید ۷ سطر داشته باشد.

    در هر سطر، اعداد از ۱ شروع می‌شوند و تا شماره همان سطر ادامه پیدا می‌کنند.

    سطر ۱: فقط عدد ۱

    سطر ۲: اعداد ۱۲

    ...

    سطر ۷: اعداد ۱۲۳۴۵۶۷

    پاسخ 
    #include <iostream>

using namespace std;

int main() {
    const int num_rows = 7;
    cout << "--- مثلث عددی افزایشی (" << num_rows << " سطر) ---\n";

    // حلقه بیرونی برای سطرها (i از 1 تا 7)
    // متغیر i شماره سطر فعلی را نشان می دهد
    for (int i = 1; i <= num_rows; ++i) {
        // حلقه داخلی برای چاپ اعداد در هر سطر (j از 1 تا i)
        // اعداد از 1 شروع می شوند و تا شماره سطر (i) ادامه پیدا می کنند
        for (int j = 1; j <= i; ++j) {
            // چاپ عدد j
            cout << j;
        }
        // رفتن به سطر بعدی پس از اتمام چاپ اعداد سطر فعلی
        cout << endl;
    }

    return 0;
}
